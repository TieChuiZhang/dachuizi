通过CocoPods集成

在Podfile中加入

pod 'UMengSocialCOM'          # 友盟分享组件

打开终端 

cd [存放Podfile的项目路径]
pod install

URL Scheme写法

AppDelegate中代码

// 设置友盟 APP KEY
[UMSocialData setAppKey:@""];

// 新浪微博设置
[UMSocialSinaSSOHandler openNewSinaSSOWithAppKey:sinaAppKey secret:sinaSecret RedirectURL:sinaRedirectURI];

//腾讯qq设置
[UMSocialQQHandler setQQWithAppId:qqAPPID appKey:qqAPPKey url:nil];
